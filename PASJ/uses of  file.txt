		CREATE DATABASE case_management;

		CREATE TABLE cases (
    		case_id INT AUTO_INCREMENT PRIMARY KEY,
    		complainant_id VARCHAR(50) NOT NULL,
    		case_type VARCHAR(50) NOT NULL,
    		date_time DATETIME NOT NULL,
    		evidence TEXT NOT NULL,
    		description TEXT NOT NULL
		);

		ALTER TABLE cases
		ADD COLUMN case_status text;

		ALTER TABLE cases
		ADD COLUMN evidance2 text;

		


		CREATE TABLE prisoners (
    		id INT AUTO_INCREMENT PRIMARY KEY,
    		nic VARCHAR(20) NOT NULL,
    		name VARCHAR(100) NOT NULL,
    		birthday DATE NOT NULL,
    		date_imprisoned DATE NOT NULL,
    		period_of_imprisonment VARCHAR(50) NOT NULL,
    		acquitted_on DATE,
    		responsible_crime VARCHAR(255) NOT NULL,
    		case_reference_no VARCHAR(50) NOT NULL,
    		image VARCHAR(255) NOT NULL
		);

		CREATE TABLE bookings (
    		id INT AUTO_INCREMENT PRIMARY KEY,
    		nic VARCHAR(12) NOT NULL,
    		name VARCHAR(100) NOT NULL,
    		date DATE NOT NULL,
    		time TIME NOT NULL
		);


		CREATE TABLE police_profiles (
    		id INT AUTO_INCREMENT PRIMARY KEY,
    		police_id VARCHAR(50) NOT NULL,
    		nic VARCHAR(20) NOT NULL UNIQUE,
   		name VARCHAR(100) NOT NULL,
   		mobile VARCHAR(15),
    		birthday DATE,
    		address TEXT,
    		date_joined DATE,
    		date_leaving DATE,
    		position VARCHAR(50),
    		promotions TEXT
);
